# Quickstart: Simplify Constraint API & Deep Attribute Type Safety

## New buildConstraints Usage

```ts
import { Toride, ForbiddenError } from "toride";
import type { GeneratedSchema } from "./generated-types.js";

const toride = new Toride<GeneratedSchema>({ policy });

// buildConstraints returns { ok, constraint } instead of the old 3-way union
const result = await toride.buildConstraints(actor, "read", "Document");

if (!result.ok) {
  // Forbidden — actor cannot read any documents
  throw new ForbiddenError(actor, "read", "Document");
}

if (result.constraint === null) {
  // Unrestricted — actor can read all documents, no WHERE clause needed
  return db.document.findMany();
}

// Constrained — translate to a query
const where = toride.translateConstraints(result.constraint, adapter);
return db.document.findMany({ where });
```

## Nested Attribute Declarations in YAML

```yaml
version: "1"

actors:
  User:
    attributes:
      email: string
      level: number
      active: boolean

resources:
  Document:
    roles: [viewer, editor, admin]
    permissions: [read, write, delete]
    attributes:
      status: string                    # flat (unchanged)
      tags: string[]                    # primitive array
      metadata:                         # nested object
        priority: string
        category: string
      collaborators:                    # array of objects
        type: array
        items:
          userId: string
          role: string
    relations:
      org: Organization
    grants:
      viewer: [read]
      editor: [read, write]
      admin: [all]
    derived_roles:
      - role: editor
        actor_type: User
        when:
          "$resource.metadata.priority": "high"   # dot-path into nested object
```

## Typed Resolvers

```ts
import { createToride } from "toride";
import type { GeneratedSchema } from "./generated-types.js";

const toride = createToride<GeneratedSchema>({
  policy,
  resolvers: {
    // TypeScript enforces the return type matches Document attributes
    Document: async (ref) => {
      const doc = await db.document.findUnique({ where: { id: ref.id } });
      return {
        status: doc.status,
        tags: doc.tags,
        metadata: doc.metadata,           // nested object
        collaborators: doc.collaborators,  // array of objects
      };
    },
  },
});
```
